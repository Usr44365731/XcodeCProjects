#include <stdio.h>
#include <ctype.h>

int main (void) {
    
   //I, V, X, L, C, D, M
    
    char a;
    int b;
    printf("Podaj liczbę rzymską: ");
    scanf("%c", &a);
    
    if (a == 'I') {
        b = 1;
    }
    else if (a == 'V') {
        b = 5;
    }
    else if (a == 'X') {
        b = 10;
    }
    else if (a == 'L') {
        b = 50;
    }
    else if (a == 'C') {
        b = 100;
    }
    else if (a == 'D') {
        b = 500;
    }
    else if (a == 'M') {
        b = 10000;
    }
    else if (isalpha(a) == 1 && (a != 'I' || a != 'V' || a != 'X' || a != 'L' || a != 'C' || a != 'D' || a != 'M')) {
        
        printf("Podano nie poprawną liczbe\n");
    }
    else if (isalpha(a) == 0) {
        
        printf("Podano niepoprawne dane\n");
        
    }
    
    printf("%d\n", b);
    
    return 0;
}
